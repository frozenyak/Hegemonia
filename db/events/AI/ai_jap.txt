# Japan invades China from the North
#############################################
event = {
	id = 3011007 
	country = JAP
	persistent = yes
	
	trigger = {
		ai = yes
		NOT = { local_flag = ChinaNorth }
		war = { country = CYN }
		NOT = { war = { country = MEX } }
		NOT = { war = { country = ENG } }
		NOT = { war = { country = RUS } }
		OR = {
			NOT = { control = { province = 1331 data = -3 } }
			NOT = { control = { province = 1332 data = -3 } }
			NOT = { control = { province = 1333 data = -3 } }
			NOT = { control = { province = 1334 data = -3 } }
		}
		NOT = { # Southern Invasion
			NOT = { ai = CYN }
			AND = {
				NOT = { control = { province = 1365 data = -3 } }
				OR = {
					NOT = { garrison = { country = CYN province = 1365 type = land size = 1 area = no } } 
					AND = {
						NOT = { ai = CYN }
						NOT = { garrison = { country = CYN province = 1365 type = land size = 2 area = no } } 
					}
				}
			}
			AND = {
				war = { country = CYN }
				NOT = { control = { province = 1365 data = -3 } }
				OR = {
					NOT = { garrison = { country = CYN province = 1365 type = land size = 1 area = no } } 
					AND = {
						NOT = { ai = CYN }
						NOT = { garrison = { country = CYN province = 1365 type = land size = 2 area = no } }
					}
				}
			}
			AND = {
				NOT = { control = { province = 1367 data = -3 } }
				OR = {
					NOT = { garrison = { country = CYN province = 1367  type = land size = 1 area = no } }
					AND = {
						NOT = { ai = CYN }
						NOT = { garrison = { country = CYN province = 1367  type = land size = 2 area = no } }
					}
				}
			}
			AND = {
				war = { country = CYN }
				NOT = { control = { province = 1367 data = -3 } }
				OR = {
					NOT = { garrison = { country = CYN province = 1367  type = land size = 1 area = no } }
					AND = {
						NOT = { ai = CYN }
						NOT = { garrison = { country = CYN province = 1367  type = land size = 2 area = no } }
					}
				}
			}
		}
	}

	name = "AI_EVENT"
	style = 0	
	
	date = { day = 1 month = january year = 1933 }
	offset = 4
	deathdate = { day = 29 month = december year = 1963 }

	action = {
		command = { type = local_clrflag which = ChinaSouth }
		command = { type = local_setflag which = ChinaNorth }
		command = { type = ai which = "switch/JAP_China_North.ai" }
	}
}
